private
  variable
    ℓ : Level
    A : Type ℓ
    a : A

Ω Ω² Ω³ Ω⁴ : (A : Type ℓ) → A → Type ℓ
Ω A a = Path A a a
Ω² A a = Ω (Ω A a) refl
Ω³ A a = Ω (Ω² A a) refl
Ω⁴ A a = Ω (Ω³ A a) refl

data JS¹ : Type where
  base : JS¹
  loops : (x : JS¹) → Path JS¹ x x

data J₃S¹ : Type where
  base : J₃S¹
  loop : PathP (λ i → J₃S¹) base base
  loop₂ : PathP (λ i → PathP (λ j → J₃S¹)
                             (loop i)
                             (loop i))
                (λ j → loop j)
                (λ j → loop j)
  loop₃ : PathP (λ i → PathP (λ j → PathP (λ k → J₃S¹)
                                          (loop₂ i j)
                                          (loop₂ i j))
                             (λ k → loop₂ i k)
                             (λ k → loop₂ i k))
                (λ j k → loop₂ j k)
                (λ j k → loop₂ j k)


loop₂' : PathP (λ i → PathP (λ j → J₃S¹)
                             (loop i)
                             (loop i))
                (λ j → loop j)
                (λ j → loop j)
loop₂' i j = loop₂ i j

csq : ∀ {ℓ} {A : Type ℓ} {x y z : A} (p : x ≡ y) (q : y ≡ z)
  → Path (y ≡ y) refl refl → PathP (λ i → p i ≡ q i) p q
csq p q r i j =
  hcomp (λ k → λ { (i = i0) → p (~ k ∨ j)
                 ; (i = i1) → q (k ∧ j)
                 ; (j = i0) → p (~ k ∨ i)
                 ; (j = i1) → q (k ∧ i)
                 })
        (r i j)

csqInv : ∀ {ℓ} {A : Type ℓ} {x y z : A} (p : x ≡ y) (q : y ≡ z)
  → PathP (λ i → p i ≡ q i) p q → Path (y ≡ y) refl refl
csqInv p q r i j =
  hcomp (λ k → λ { (i = i0) → p (k ∨ j)
                 ; (i = i1) → q (~ k ∧ j)
                 ; (j = i0) → p (k ∨ i)
                 ; (j = i1) → q (~ k ∧ i)
                 })
        (r i j)

hsqInv : {x y : A} (p : x ≡ y) → p ≡ p → Ω² A y
hsqInv {y = y} p r i j =
  hcomp (λ k → λ { (i = i0) → p (j ∨ k)
                 ; (i = i1) → p (j ∨ k)
                 ; (j = i0) → p k
                 ; (j = i1) → y
                 })
        (r i j)

gen-π₂JS¹ : Ω² JS¹ base
gen-π₂JS¹ = csqInv (loops base) (loops base) (λ i j → loops (loops base j) i)

gen-π₂J₃S¹ : Ω² J₃S¹ base
gen-π₂J₃S¹ = csqInv loop loop loop₂


-- Wedge-y stuff

data JS¹∨S¹ : Type where
  base : JS¹∨S¹
  loops : Bool → (x : JS¹∨S¹) → Path JS¹∨S¹ x x

data J₃S¹∨S¹ : Type where
  base : J₃S¹∨S¹
  loop : Bool → PathP (λ i → J₃S¹∨S¹) base base
  loop₂ : (b₀ b₁ : Bool) → 
    PathP (λ i → PathP (λ j → J₃S¹∨S¹)
                       (loop b₀ i)
                       (loop b₀ i))
          (λ j → loop b₁ j)
          (λ j → loop b₁ j)
  loop₃ : (b₀ b₁ b₂ : Bool) → 
    PathP (λ i → PathP (λ j → PathP (λ k → J₃S¹∨S¹)
                                    (loop₂ b₀ b₁ i j)
                                    (loop₂ b₀ b₁ i j))
                       (λ k → loop₂ b₀ b₂ i k)
                       (λ k → loop₂ b₀ b₂ i k))
          (λ j k → loop₂ b₁ b₂ j k)
          (λ j k → loop₂ b₁ b₂ j k)


gen-π₂JS¹∨S¹ : (x : JS¹∨S¹) → Ω² JS¹∨S¹ x
gen-π₂JS¹∨S¹ x =
  hsqInv (loops true x)
    (λ i j → hcomp (λ k → λ { (i = i0) → loops true x j
                            ; (i = i1) → loops true (loops false x (~ k)) j
                            ; (j = i0) → loops false x (i ∧ ~ k)
                            ; (j = i1) → loops false x (i ∧ ~ k)
                            })
                   (loops false (loops true x j) i))

gen-π₂JS¹∨S¹-base : Ω² JS¹∨S¹ base
gen-π₂JS¹∨S¹-base = gen-π₂JS¹∨S¹ base

gen-π₂J₃S¹∨S¹ : Ω² J₃S¹∨S¹ base
gen-π₂J₃S¹∨S¹ =
  hsqInv (loop true)
    (λ i j → hcomp (λ k → λ { (i = i0) → loop true j
                            ; (i = i1) → loop₂ true false j (~ k)
                            ; (j = i0) → loop false (i ∧ ~ k)
                            ; (j = i1) → loop false (i ∧ ~ k)
                            })
                   (loop₂ false true i j))
