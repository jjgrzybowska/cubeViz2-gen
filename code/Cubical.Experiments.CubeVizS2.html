module Cubical.Experiments.CubeVizS2 where

open import Cubical.Foundations.Prelude
open import Cubical.HITs.S2.Base

erp : I → I → I → I
erp t x y = (~ t ∧ x) ∨ (t ∧ y) ∨ (x ∧ y)

variable
  ℓ : Level
  A : Type ℓ
  a : A

Ω Ω² Ω³ : (A : Type ℓ) → A → Type ℓ
Ω A a = Path A a a
Ω² A a = Path (Ω A a) refl refl
Ω³ A a = Path (Ω² A a) refl refl

csqInv : ∀ {ℓ} {A : Type ℓ} {x y z : A} (p : x ≡ y) (q : y ≡ z)
  → PathP (λ i → p i ≡ q i) p q → Path (y ≡ y) refl refl
csqInv p q r i j =
  hcomp (λ k → λ { (i = i0) → p (k ∨ j)
                 ; (i = i1) → q (~ k ∧ j)
                 ; (j = i0) → p (k ∨ i)
                 ; (j = i1) → q (~ k ∧ i)
                 })
        (r i j)

hsqInv : {x y : A} (p : x ≡ y) → p ≡ p → Ω² A y
hsqInv {y = y} p r i j =
  hcomp (λ k → λ { (i = i0) → p (j ∨ k)
                 ; (i = i1) → p (j ∨ k)
                 ; (j = i0) → p k
                 ; (j = i1) → y
                 })
        (r i j)

one' : PathP (λ i → surf i ≡ surf i) surf surf
one' i j k =
  hcomp (λ l → λ { (i = i0) → surf j (k ∧ l)
                 ; (i = i1) → surf j (k ∧ l)
                 ; (j = i0) → surf i k
                 ; (j = i1) → surf i k
                 ; (k = i0) → base
                 ; (k = i1) → surf j l
                 })
        (surf i k)

one : Ω³ S² base
one i j k = csqInv surf surf one' i j k

-- surf with some extra space around it
s-shrunk : Ω² S² base
s-shrunk i j =
  hcomp (λ l → λ { (i = i0) → base
                 ; (i = i1) → base
                 ; (j = i0) → base
                 ; (j = i1) → base
                 })
        (surf i j)

one'-shrunk : PathP (λ i → s-shrunk i ≡ s-shrunk i) s-shrunk s-shrunk
one'-shrunk i j k =
  hcomp (λ l → λ { (i = i0) → s-shrunk j (k ∧ l)
                 ; (i = i1) → s-shrunk j (k ∧ l)
                 ; (j = i0) → s-shrunk i k
                 ; (j = i1) → s-shrunk i k
                 ; (k = i0) → base
                 ; (k = i1) → s-shrunk j l
                 })
        (s-shrunk i k)

one-shrunk : Ω³ S² base
one-shrunk i j k = csqInv s-shrunk s-shrunk one'-shrunk i j k

-- symmetric version of one' (w/ over and under bumps)
oneSym' : PathP (λ i → surf i ≡ surf i) surf surf
oneSym' i j k =
  hcomp (λ f → λ { (i = i0) → surf j (k ∨ ~ f)
                 ; (i = i1) → surf j (k ∨ ~ f)
                 ; (j = i0) → surf i (k ∧ f)
                 ; (j = i1) → surf i (k ∧ f)
                 ; (k = i0) → surf j (~ f)
                 ; (k = i1) → surf i f
                 })
        base

oneSym : Ω³ S² base
oneSym i j k = csqInv surf surf oneSym' i j k

brunerie' : surf ≡ surf
brunerie' j a b =
  hcomp (λ i → λ { (j = i0) → surf a b
                 ; (j = i1) → surf a b
                 ; (a = i0) → surf i j
                 ; (a = i1) → surf i j
                 ; (b = i0) → surf i j
                 ; (b = i1) → surf i j
                 })
        (surf a b)

brunerie : Ω³ S² base
brunerie i j k = hsqInv surf brunerie' i j k

brunerieTurned : surf ≡ surf
brunerieTurned j a b =
  hcomp (λ i → λ { (j = i0) → oneSym' i a b
                 ; (j = i1) → oneSym' a i b
                 ; (a = i0) → surf i b
                 ; (a = i1) → surf i b
                 ; (b = i0) → base
                 ; (b = i1) → base
                 })
        (surf a b)

brunerie'-shrunk : s-shrunk ≡ s-shrunk
brunerie'-shrunk j a b =
  hcomp (λ i → λ { (j = i0) → s-shrunk a b
                 ; (j = i1) → s-shrunk a b
                 ; (a = i0) → s-shrunk i j
                 ; (a = i1) → s-shrunk i j
                 ; (b = i0) → s-shrunk i j
                 ; (b = i1) → s-shrunk i j
                 })
        (s-shrunk a b)

brunerie-shrunk : Ω³ S² base
brunerie-shrunk i j k = hsqInv s-shrunk brunerie'-shrunk i j k

rot90 : surf ≡ (λ i j → surf j (~ i))
rot90 t i j =
  hcomp (λ l → λ { (t = i0) → surf i j
                 ; (t = i1) → surf j (~ i)
                 ; (i = i0) → surf (erp l (erp t i0 j) i0) (erp t j i1)
                 ; (i = i1) → surf (erp l (erp t i1 j) i1) (erp t j i0)
                 ; (j = i0) → surf (erp l (erp t i i0) i0) (erp t i0 (~ i))
                 ; (j = i1) → surf (erp l (erp t i i1) i1) (erp t i1 (~ i))
                 })
        (surf (erp t i j) (erp t j (~ i)))

twisted : Ω³ S² base
twisted i j k =
  hcomp (λ l → λ { (i = i0) → rot90 j l k
                 ; (i = i1) → rot90 (~ j) (~ l) (~ k)
                 ; (j = i0) → rot90 (~ i) (~ k) l
                 ; (j = i1) → rot90 i k (~ l)
                 ; (k = i0) → base
                 ; (k = i1) → base
                 })
        base

rot90-shrunk1 : Path (Ω² S² base) s-shrunk (λ i j → s-shrunk j (~ i))
rot90-shrunk1 t i j =
  hcomp (λ l → λ { (i = i0) → base
                 ; (i = i1) → base
                 ; (j = i0) → base
                 ; (j = i1) → base
                 })
        (rot90 t i j)

twisted-shrunk1 : Ω³ S² base
twisted-shrunk1 i j k =
  hcomp (λ l → λ { (i = i0) → rot90-shrunk1 j l k
                 ; (i = i1) → rot90-shrunk1 (~ j) (~ l) (~ k)
                 ; (j = i0) → rot90-shrunk1 (~ i) (~ k) l
                 ; (j = i1) → rot90-shrunk1 i k (~ l)
                 ; (k = i0) → base
                 ; (k = i1) → base
                 })
        base
