private
  variable
    ℓ : Level
    A : Type ℓ
    x : A

dsqInv : {x y : A} (p : x ≡ y) → p ≡ p → Ω² A x
dsqInv {x = x} p r i j =
  hcomp (λ k → λ { (i = i0) → p (j ∧ ~ k)
                 ; (i = i1) → p (j ∧ ~ k)
                 ; (j = i0) → x
                 ; (j = i1) → p (~ k)
                 })
        (r i j)

W₂₂'-turned2D : Path (Ω J₃S¹ base) loop loop
W₂₂'-turned2D j a =
  hcomp (λ i → λ { (j = i0) → loop₂ a i
                 ; (j = i1) → loop₂ i a
                 ; (a = i0) → loop i
                 ; (a = i1) → loop i
                 })
        (loop a)

b1 : Ω² J₃S¹ base
b1 = dsqInv loop W₂₂'-turned2D

hopf2D : Ω² J₃S¹ base
hopf2D = csqInv loop loop loop₂

b2 : Ω² J₃S¹ base
b2 = dsqInv loop (λ i j → hcomp (λ k → λ { (i = i0) → csq loop loop hopf2D j k
                                         ; (i = i1) → csq loop loop hopf2D k j
                                         ; (j = i0) → loop k
                                         ; (j = i1) → loop k
                                         })
                                (loop j))

b3 : Ω² J₃S¹ base
b3 = dsqInv loop (λ i j → hcomp (λ k → λ { (i = i0) → csq refl loop hopf2D j k
                                         ; (i = i1) → csq refl loop hopf2D k j
                                         ; (j = i0) → base
                                         ; (j = i1) → loop k
                                         })
                                base)

b4 : Ω² J₃S¹ base
b4 = dsqInv refl (λ i j → hcomp (λ k → λ { (i = i0) → csq refl refl hopf2D j k
                                      ; (i = i1) → csq refl refl hopf2D k j
                                      ; (j = i0) → base
                                      ; (j = i1) → base
                                      })

b5 : Ω² J₃S¹ base
b5 = (λ i j → hcomp (λ k → λ { (i = i0) → csq refl refl hopf2D j k
                          ; (i = i1) → csq refl refl hopf2D k j

b6 : Ω² J₃S¹ base
