Ω Ω² Ω³ Ω⁴ : (A : Type ℓ) → A → Type ℓ
Ω A a = Path A a a
Ω² A a = Ω (Ω A a) refl
Ω³ A a = Ω (Ω² A a) refl
Ω⁴ A a = Ω (Ω³ A a) refl

csqInv : ∀ {ℓ} {A : Type ℓ} {x y z : A} (p : x ≡ y) (q : y ≡ z)
  → PathP (λ i → p i ≡ q i) p q → Path (y ≡ y) refl refl
csqInv p q r i j =
  hcomp (λ k → λ { (i = i0) → p (k ∨ j)
                 ; (i = i1) → q (~ k ∧ j)
                 ; (j = i0) → p (k ∨ i)
                 ; (j = i1) → q (~ k ∧ i)
                 })
        (r i j)

hsqInv : {x y : A} (p : x ≡ y) → p ≡ p → Ω² A y
hsqInv {y = y} p r i j =
  hcomp (λ k → λ { (i = i0) → p (j ∨ k)
                 ; (i = i1) → p (j ∨ k)
                 ; (j = i0) → p k
                 ; (j = i1) → y
                 })
        (r i j)


erp : I → I → I → I
erp t x y = (~ t ∧ x) ∨ (t ∧ y) ∨ (x ∧ y)

-- Eckmann-Hilton (symmetric version)
EH : (p q : Ω² A a) → PathP (λ i → PathP (λ j → Ω A a) (p i) (p i)) q q
EH {a = a} p q i j k =
  hcomp (λ f → λ { (i = i0) → q j (erp f i1 k)
                 ; (i = i1) → q j (erp f i1 k)
                 ; (j = i0) → p i (erp f i0 k)
                 ; (j = i1) → p i (erp f i0 k)
                 ; (k = i0) → q j (~ f)
                 ; (k = i1) → p i f
                 })
         a

-- Whitehead product
W₂₂ : Ω² A a → Ω² A a → Ω³ A a
W₂₂ p q =
  hsqInv q
    (λ j a b → hcomp (λ i → λ { (j = i0) → q a b
                              ; (j = i1) → q a b
                              ; (a = i0) → p i j
                              ; (a = i1) → p i j
                              ; (b = i0) → p i j
                              ; (b = i1) → p i j
                              })
                     (q a b))

csqPath : ∀ {ℓ} {A : Type ℓ} {x y z : A} (p : x ≡ y) (q : y ≡ z)
  → Path (y ≡ y) refl refl ≡ PathP (λ i → p i ≡ q i) p q
csqPath {A = A} p q k =
  PathP (λ i → PathP (λ j → A)
                     (p (~ k ∨ i))
                     (q (k ∧ i)))
        (λ j → p (~ k ∨ j))
        (λ j → q (k ∧ j))

constCube : PathP (λ i → PathP (λ j → PathP (λ k → S²) (surf i j) (surf i j)) (λ k → surf i k) (λ k → surf i k)) (λ j k → surf j k) (λ j k → surf j k)
constCube i = transp (λ t → csqPath (surf i) (surf i) t) (~ i ∨ i) surf
